taizu_buff1_start = {
	PRC = {
		add_ideas = PRC_taizu_buff1
	}
	add_to_variable = { 
		var = taizu_var_cost
		value = taizu_var_cost1
	}
	add_to_variable = { show_taizu_menu1 = 1}
}
taizu_buff1_end = {
	PRC = {
		remove_ideas = PRC_taizu_buff1
	}
	subtract_from_variable= { 
		var = taizu_var_cost
		value = taizu_var_cost1
	}
	subtract_from_variable = { show_taizu_menu1 = 1}
}
