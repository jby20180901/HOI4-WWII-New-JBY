add_namespace = gf

country_event = {
	id = gf.1 #陆军
	title = tech.1.t
	desc = tech.1.d
	picture = GFX_report_event_soldiers_marching
	
    trigger = {
        if = {
            limit = {
                OR = {
                    AND = {
                        has_tech = mobile_warfare
                        NOT = {
                            has_tech = werwolf_guerillas
                            has_tech = modern_blitzkrieg
                        }
                    }
                    AND = {
                        has_tech = superior_firepower
                        NOT = {
                            has_tech = air_land_battle
                            has_tech = shock_and_awe
                        }
                    }
                    AND = {
                        has_tech = trench_warfare
                        NOT = {
                            has_tech = c3i_theory
                            has_tech = infiltration_in_depth
                        }
                    }
                    AND = {
                        has_tech = mass_assault
                        NOT = {
                            has_tech = continuous_offensive
                            has_tech = guerilla_warfare
                        }
                    }
                }
            }
            always = no
        }
        if = {
            limit = {
                AND = {
                    NOT = { has_tech = mobile_warfare }
                    NOT = { has_tech = superior_firepower }
                    NOT = { has_tech = trench_warfare }
                    NOT = { has_tech = mass_assault }
                }
            }
            always = no
        }
        OR = {
            NOT = { has_tech = mobile_warfare }
			NOT = { has_tech = superior_firepower }
			NOT = { has_tech = trench_warfare }
			NOT = { has_tech = mass_assault }
        }
    }

    option = {
        name = tech.1.a
        ai_chance = { factor = 25 }
        set_technology = {
            mobile_warfare = 1
        }
    }
    option = {
        name = tech.1.b
        ai_chance = { factor = 25 } 
        set_technology = {
            superior_firepower = 1
        }
    }
    option = {
        name = tech.1.c
        ai_chance = { factor = 25 } 
        set_technology = {
            trench_warfare = 1
        }
    }
    option = {
        name = tech.1.e
        ai_chance = { factor = 25 } 
        set_technology = {
            mass_assault = 1
        }
    }
}
