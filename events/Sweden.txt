		NOT = {

				is_in_faction_with = GER
				GER = {has_non_aggression_pact_with = SOV}
			}
		}
		NOT = { is_in_faction_with = SOV }
	}
	
	option = { # Join the winter war

		country_event = { days = 5 id = finland.0 }
		
		
		
	}
    option = { # Send equipment

        send_equipment = {
			equipment = infantry_equipment
			amount = 5000
			target = FIN
		}
		if = {
			limit = {
				has_idea = SWE_Affirmed_Neutrality
			}
			swap_ideas = {
				remove_idea = SWE_Affirmed_Neutrality
				add_idea = SWE_Less_Neutral
			}
		}
		else_if = {
			limit = {
				has_idea = neutrality_idea
			}

				add_idea = SWE_Less_Neutral
			}
		}
